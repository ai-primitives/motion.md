# Interactive Elements

export const InteractiveDemo = () => {
  const [active, setActive] = useState(false);
  return (
    <div
      style={{
        padding: '2rem',
        background: active ? '#4CAF50' : '#2196F3',
        transition: 'background 0.3s',
        cursor: 'pointer'
      }}
      onClick={() => setActive(!active)}
    >
      Click to toggle state
    </div>
  );
};
